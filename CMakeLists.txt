cmake_minimum_required(VERSION 3.2)

PROJECT(luastroke)
INCLUDE(cmake/ModuleImport.cmake)
INCLUDE(cmake/LuaModule.cmake)
LIST(APPEND CMAKE_MODULE_PATH ${CMAKE_CURRENT_SOURCE_DIR}/cmake)
INCLUDE(cmake/ModuleCompileOptions.cmake)
ModuleSetCompileOptions()
ModuleSetWinCompilerFlags()

SET(CMAKE_CXX_STANDARD 17)
SET(CMAKE_CXX_STANDARD_REQUIRED ON)

SET(LUA_MODULE lua53)
SET(LUASTROKE_VERSION_MAJOR "1")
SET(LUASTROKE_VERSION_MINOR "0")
SET(LUASTROKE_VERSION_PATCH "1")
SET(LUASTROKE_VERSION "${LUASTROKE_VERSION_MAJOR}.${LUASTROKE_VERSION_MINOR}.${LUASTROKE_VERSION_PATCH}")

MESSAGE(STATUS "VERSION: ${LUASTROKE_VERSION}")

ADD_DEFINITIONS(-DUSE_LUASTROKE)
ADD_DEFINITIONS(-DLUA_COMPAT_MODULE -DLUA_COMPAT_APIINTCASTS)

LuaModuleImport("${LUA_MODULE}" "luastroke" "src/luastroke" "")
